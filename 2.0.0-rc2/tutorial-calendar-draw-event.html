            var cal = new tui.component.Calendar('#layer');

        cal.on('draw', function(event) {
            console.log(event.date);
            console.log(event.layerType);
            event.$dateElements.each(function() {
                var $el = $(this);
                var date = new Date($el.data('timestamp'));
                console.log(date);
            });
        });
        cal.draw({
            date: new Date(2016, 11)
        });
