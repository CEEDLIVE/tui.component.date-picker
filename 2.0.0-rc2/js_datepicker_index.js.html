'use strict';

var Calendar = require('../calendar');
var DateFormatter = require('../dateTimeFormatter');
var RangeModel = require('./rangeModel');
var Timepicker = require('../timepicker');
var constants = require('../constants');
var localeTexts = require('../localeTexts');
var dateUtil = require('../dateUtil');
var setMouseTouchEvent = require('../setMouseTouchEvent');
var tmpl = require('../../template/datepicker/index.hbs');

var util = tui.util;

var DEFAULT_LANGUAGE_TYPE = constants.DEFAULT_LANGUAGE_TYPE;
var TYPE_DATE = constants.TYPE_DATE;
var TYPE_MONTH = constants.TYPE_MONTH;
var TYPE_YEAR = constants.TYPE_YEAR;
var CLASS_NAME_NEXT_YEAR_BTN = constants.CLASS_NAME_NEXT_YEAR_BTN;
var CLASS_NAME_NEXT_MONTH_BTN = constants.CLASS_NAME_NEXT_MONTH_BTN;
var CLASS_NAME_PREV_YEAR_BTN = constants.CLASS_NAME_PREV_YEAR_BTN;
var CLASS_NAME_PREV_MONTH_BTN = constants.CLASS_NAME_PREV_MONTH_BTN;

var CLASS_NAME_SELECTABLE = 'tui-is-selectable';
var CLASS_NAME_BLOCKED = 'tui-is-blocked';
var CLASS_NAME_SELECTED = 'tui-is-selected';
var CLASS_NAME_CHECKED = 'tui-is-checked';
var CLASS_NAME_SELECTOR_BUTTON = 'tui-datepicker-selector-button';

var SELECTOR_BODY = '.tui-datepicker-body';
var SELECTOR_FOOTER = '.tui-datepicker-footer';
var SELECTOR_DATE_ICO = '.tui-ico-date';

/**
 * Merge default option
 * @ignore
 * @param {object} option - Datepicker option
 * @returns {object}
 */
var mergeDefaultOption = function(option) {
    option = util.extend({
        language: DEFAULT_LANGUAGE_TYPE,
        calendar: {},
        inlineTimepicker: false,
        timepicker: null,
        input: {
            element: null,
            dateFormat: null
        },
        date: null,
        showAlways: false,
        type: TYPE_DATE,
        selectableRanges: [[constants.MIN_DATE, constants.MAX_DATE]],
        openers: []
    }, option);

    option.localeText = localeTexts[option.language];
    if (!util.isObject(option.calendar)) {
        throw new Error('Calendar option must be an object');
    }
    if (!util.isObject(option.input)) {
        throw new Error('Input option must be an object');
    }
    if (!util.isArray(option.selectableRanges)) {
        throw new Error('Selectable-ranges must be an 2d-array');
    }

    // override calendar option
    option.calendar.language = option.language;
    option.calendar.type = option.type;

    return option;
};

/**
 * @Class
 * @param {HTMLElement|jQuery|string} container - Container element of datepicker
 * @param {Object} option - Options
 *      @param {Date|number} [option.date] - Initial date. Default - null for no initial date
 *      @param {string} [option.language='en'] - Language key
 *      @param {HTMLElement|string|jQuery} option.input.element - Input element
 *      @param {string} [option.intput.format = 'yyyy-mm-dd'] - Date string format
 *      @param {Array.<Array.<Date|number>>} [options.selectableRanges = 1900/1/1 ~ 2999/12/31]
 *                                                                      - Selectable date ranges.
 *      @param {Array} [option.openers = []] - Opener button list (example - icon, button, etc.)
 *      @param {boolean} [option.showAlways = false] - Whether the datepicker shows always
 *      @param {Timepicker} [option.timePicker] - Timepicker option
 * @tutorial datepicker
 * @example
 *   tui.component.Datepicker.localTexts['customKey'] = {
 *       titles: {
 *           // days
 *           DD: ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Firday', 'Saturday'],
 *           // daysShort
 *           D: ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fir', 'Sat'],
 *           // months
 *           MMMM: [
 *               'January', 'February', 'March', 'April', 'May', 'June',
 *               'July', 'August', 'September', 'October', 'November', 'December'
 *           ],
 *           // monthsShort
 *           MMM: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec']
 *       },
 *       titleForamt: 'MMM yyyy',
 *       todayFormat: 'D, MMMM dd, yyyy',
 *       date: 'Date',
 *       time: 'Time'
 *   };
 *
 *   var range1 = [new Date(2015, 2, 1), new Date(2015, 3, 1)];
 *   var range2 = [1465570800000, 1481266182155];
 *
 *   var picker1 = new tui.component.Datepicker('#datepicker-container', {
 *       language: 'customKey',
 *       calendar: {
 *          hasTodayBox: true
 *       },
 *       timePicker: {
 *           showMeridiem: true,
 *           defaultHour: 13,
 *           defaultMinute: 24
 *       },
 *       input: {
 *           element: '#pickerInput',
 *           format: 'yyyy년 mm월 dd일'
 *       }
 *       type: 'date',
 *       date: new Date(2015, 0, 1) // or timestamp, default - null(no initial date)
 *       selectableRanges: [range1, range2],
 *       openers: ['#opener']
 *   });
 */
var DatePicker = util.defineClass(/** @lends DatePicker.prototype */{
    static: {
        localeTexts: localeTexts
    },
    init: function(container, option) {
        option = mergeDefaultOption(option);

        /**
         * Datepicker container
         * @type {jQuery}
         * @private
         */
        this._$container = $(container);

        /**
         * Datepicker element
         * @type {jQuery}
         * @private
         */
        this._$element = $(tmpl(option)).appendTo(this._$container);

        /**
         * Element for displaying a date value
         * @type {jQuery}
         * @private
         */
        this._$input = $();

        /**
         * Calendar instance
         * @type {Calendar}
         * @private
         */
        this._calendar = new Calendar(this._$element.find(SELECTOR_BODY), option.calendar);

        /**
         * Object having date values
         * @type {Date}
         * @private
         */
        this._date = null;

        /**
         * Selectable date-ranges model
         * @type {RangeModel}
         * @private
         */
        this._rangeModel = null;

        /**
         * Timepicker instance
         * @type {Timepicker}
         * @private
         */
        this._timepicker = null;

        /**
         * openers - opener list
         * @type {Array}
         * @private
         */
        this._openers = [];

        /**
         * State of picker enable
         * @type {boolean}
         * @private
         */
        this._isEnabled = true;

        /**
         * Date formatter
         * @private
         * @type {DateFormatter}
         */
        this._formatter = null;

        /**
         * ID of instance
         * @private
         * @type {number}
         */
        this._id = 'datepicker' + util.stamp(this);

        this._initializeDatePicker(option);
    },

    /**
     * Initialize method
     * @param {Object} option - user option
     * @private
     */
    _initializeDatePicker: function(option) {
        this.setRanges(option.selectableRanges);
        this._setEvents(option);
        this._initTimepicker(option.timepicker, option.inlineTimepicker);
        this.setDateFormat(option.input.format || 'yyyy-MM-dd');
        this.setInput(option.input.element);
        this.setDate(option.date);

        util.forEach(option.openers, this.addOpener, this);
        if (!option.showAlways) {
            this._$element.hide();
        }
    },

    /**
     * Set events
     * @param {object} option - Constructor option
     * @private
     */
    _setEvents: function(option) {
        setMouseTouchEvent(this._$container, ['touchend', 'click'], $.proxy(this._onClickDate, this), {
            selector: '.' + CLASS_NAME_SELECTABLE,
            namespace: this._id
        });
        if (option.timepicker && !option.inlineTimepicker) {
            setMouseTouchEvent(this._$container, ['touchend', 'click'], $.proxy(this._onClickSelectorButton, this), {
                selector: '.' + CLASS_NAME_SELECTOR_BUTTON,
                namespace: this._id
            });
        }
        this.getCalendar().on('draw', this._onDrawCalendar, this);
    },

    /**
     * Set Timepicker instance
     * @param {Object} opTimepicker - Timepicker instance
     * @param {boolean} [inlineTimepicker = false] - Set inline timepicker
     * @private
     */
    _initTimepicker: function(opTimepicker, inlineTimepicker) {
        if (!opTimepicker) {
            return;
        }

        if (this._date) {
            opTimepicker.initialHour = this._date.getHours();
            opTimepicker.initialMinute = this._date.getMinutes();
        }

        if (inlineTimepicker) {
            this._timepicker = new Timepicker(this._$element.find(SELECTOR_FOOTER), opTimepicker);
        } else {
            this._timepicker = new Timepicker(this._$element.find(SELECTOR_BODY), opTimepicker);
            this._timepicker.hide();
        }

        this._timepicker.on('change', function(ev) {
            if (this._date) {
                this.setDate(this._date.setHours(ev.hour, ev.minute));
            }
        }, this);
    },

    /**
     * Selector button click handler
     * @param {jQuery.Event} ev - Event object
     * @private
     */
    _onClickSelectorButton: function(ev) {
        var btnSelector = '.' + CLASS_NAME_SELECTOR_BUTTON;
        var $selectedBtn = $(ev.target).closest(btnSelector);
        var isDate = !!$selectedBtn.find(SELECTOR_DATE_ICO).length;

        if (isDate) {
            this._calendar.show();
            this._timepicker.hide();
        } else {
            this._calendar.hide();
            this._timepicker.show();
        }
        this._$element.find(btnSelector).removeClass(CLASS_NAME_CHECKED);
        $selectedBtn.addClass(CLASS_NAME_CHECKED);
    },

    /**
     * @param {HTMLElement} element - Element
     * @returns {boolean}
     * @private
     */
    _isOpener: function(element) {
        var result = false;
        var openers = this._openers;
        var i = 0;
        var len = openers.length;

        for (; i < len; i += 1) {
            if (element === openers[i] || $.contains(openers[i], element)) {
                result = true;
                break;
            }
        }

        return result;
    },

    /**
     * Set date from string
     * @param {string} str - date string
     * @private
     */
    _setDateFromString: function(str) {
        var date;
        try {
            date = this._formatter.parse(str);

            if (this.isSelectable(date)) {
                if (this._timepicker) {
                    this._timepicker.setTime(date.getHours(), date.getMinutes());
                }
                this.setDate(date);
            } else {
                this.setDate(this._date); // Rollback input value
            }
        } catch (err) {
            this.setDate(this._date); // Rollback input value
            throw err; // Throw to user
        }
    },

    /**
     * Set selectable-class-name to selectable date element.
     * @param {jQuery} $dateElements - date element
     * @private
     */
    _setDefaultClassName: function($dateElements) {
        var self = this;

        $dateElements.each(function(idx, el) {
            var $el = $(el);
            var date = new Date($el.data('timestamp'));

            if (self.isSelectable(date)) {
                $el.addClass(CLASS_NAME_SELECTABLE);
            } else {
                $el.addClass(CLASS_NAME_BLOCKED);
            }
        });
    },

    /**
     * Set selected-class-name to selected date element
     * @param {HTMLElement|jQuery} $dateElements - date element
     * @private
     */
    _setSelectedClassName: function($dateElements) {
        var self = this;

        $dateElements.each(function(idx, el) {
            var $el = $(el);
            var date = new Date($el.data('timestamp'));

            if (self.isSelectable(date) && self.isSelected(date)) {
                $el.addClass(CLASS_NAME_SELECTED);
            } else {
                $el.removeClass(CLASS_NAME_SELECTED);
            }
        });
    },

    /**
     * Set value a date-string of current this instance to input element
     * @private
     */
    _syncToInput: function() {
        if (!this._date) {
            return;
        }

        this._$input.val(this._formatter.format(this._date));
    },

    /**
     * Event handler for mousedown of document<br>
     * - When click the out of layer, close the layer
     * @param {Event} event - event object
     * @private
     */
    _onMousedownDocument: function(event) {
        var evTarget = event.target;
        var isContains = $.contains(this._$element[0], evTarget);
        var isInput = this._$input.is(evTarget);
        var shouldClose = !(this.showAlways || isInput || isContains || this._isOpener(event.target));

        if (shouldClose) {
            this.close();
        }
    },

    /**
     * Event handler for click of calendar<br>
     * - Update date form event-target
     * @param {Event} ev - event object
     * @private
     */
    _onClickDate: function(ev) {
        var timestamp = $(ev.target).data('timestamp');
        var newDate = new Date(timestamp);
        var timepicker = this._timepicker;
        var prevDate = this._date;

        if (timepicker) {
            newDate.setHours(timepicker.getHour(), timepicker.getMinute());
        } else if (prevDate) {
            newDate.setHours(prevDate.getHours(), prevDate.getMinutes());
        }
        this.setDate(newDate);


        /**
         * 'shouldClose' event from select date from mouse-click
         * @event DatePicker#shouldClose
         * @example
         * datepicker.on('shouldClose', function() {
         *      return false; // Do not close layer
         *      // return true; // Close layer
         * });
         */
        if (!this.showAlways && this.invoke('shouldClose')) {
            this.close();
        }
    },

    /**
     * Event handler for 'draw'-custom event of calendar
     * @param {Object} eventData - custom event data
     * @see {Calendar.draw}
     * @private
     */
    _onDrawCalendar: function(eventData) {
        var $dateElements = eventData.$dateElements;

        this._setDefaultClassName($dateElements);
        if (this._date) {
            this._setSelectedClassName($dateElements);
        }

        this._hideUnneccessaryButtons();
    },

    /**
     * Hide invalid buttons (next, next-year, prev, prev-year)
     * @see Don't save buttons reference. The buttons are rerendered every "calendar.darw".
     * @private
     */
    _hideUnneccessaryButtons: function() {
        var nextYearDate = this._calendar.getNextYearDate();
        var prevYearDate = this._calendar.getPrevYearDate();
        var maxTimestamp = this._rangeModel.getMaximumValue();
        var minTimestamp = this._rangeModel.getMinimumValue();
        var nextMonthDate, prevMonthDate;

        if (maxTimestamp < nextYearDate.getTime()) {
            this._$element.find('.' + CLASS_NAME_NEXT_YEAR_BTN).hide();
        }

        if (minTimestamp > prevYearDate.getTime()) {
            this._$element.find('.' + CLASS_NAME_PREV_YEAR_BTN).hide();
        }

        if (this.getCurrentType() === TYPE_DATE) {
            nextMonthDate = this._calendar.getNextDate();
            prevMonthDate = this._calendar.getPrevDate();

            if (maxTimestamp < nextMonthDate.getTime()) {
                this._$element.find('.' + CLASS_NAME_NEXT_MONTH_BTN).hide();
            }

            if (minTimestamp > prevMonthDate.getTime()) {
                this._$element.find('.' + CLASS_NAME_PREV_MONTH_BTN).hide();
            }
        }
    },

    /**
     * Input change handler
     * @param {jQuery.Event} ev - Event object
     * @private
     */
    _onChangeInput: function(ev) {
        this._setDateFromString(ev.target.value);
    },

    /**
     * Get current datepicker type
     * @returns {string}
     */
    getCurrentType: function() {
        return this._calendar.getType();
    },

    /**
     * Whether the provided date is selectable
     * @param {Date} date - Date instance
     * @returns {boolean}
     */
    isSelectable: function(date) {
        var start, end;

        start = new Date(date);
        end = new Date(date);

        switch (this.getCurrentType()) {
            case TYPE_YEAR:
                start.setMonth(0, 1);
                end.setMonth(11, 31);
                break;
            case TYPE_MONTH:
                start.setDate(1);
                end.setMonth(end.getMonth() + 1, 0);
                break;
            default: break;
        }

        return this._rangeModel.hasOverlap(start.getTime(), end.getTime());
    },

    /**
     * Is selected date ?
     * @param {Date} date - Date instance
     * @returns {boolean}
     */
    isSelected: function(date) {
        var curDate = this._date;

        switch (this.getCurrentType()) {
            case TYPE_DATE:
                return curDate.setHours(0, 0, 0, 0) === date.setHours(0, 0, 0, 0);
            case TYPE_MONTH:
                return (curDate.getFullYear() === date.getFullYear()) && (curDate.getMonth() === date.getMonth());
            case TYPE_YEAR:
                return curDate.getFullYear() === date.getFullYear();
            default:
                return false;
        }
    },

    /**
     * Set start/end edge from selectable-ranges
     * @param {Array.<Array<Date|number>>} ranges - (2d-array) Selectable ranges
     * @private
     */
    setRanges: function(ranges) {
        ranges = tui.util.map(ranges, function(range) {
            var start = new Date(range[0]).getTime();
            var end = new Date(range[1]).getTime();

            return [start, end];
        });

        this._rangeModel = new RangeModel(ranges);
    },

    /**
     * Add a range
     * @param {Date|number} start - startDate or one day
     * @param {Date|number} end - endDate
     * @example
     * var start = new Date(2015, 1, 3);
     * var end = new Date(2015, 2, 6);
     *
     * datepicker.addRange(start, end);
     */
    addRange: function(start, end) {
        start = new Date(start).getTime();
        end = new Date(end).getTime();

        this._rangeModel.add(start, end);
        this._calendar.draw(); // view update
    },

    /**
     * Remove a range
     * @param {Date|number} start - startDate or one day
     * @param {Date|number} end - endDate
     * @example
     * var start = new Date(2015, 1, 3);
     * var end = new Date(2015, 2, 6);
     *
     * datepicker.removeRange(start, end);
     */
    removeRange: function(start, end) {
        start = new Date(start).getTime();
        end = new Date(end).getTime();

        this._rangeModel.exclude(start, end);
        this._calendar.draw(); // view update
    },

    /**
     * add opener
     * @param {HTMLElement|jQuery|string} opener - element or selector
     */
    addOpener: function(opener) {
        var $opener = $(opener);

        opener = $opener[0];

        if (opener && util.inArray(opener, this._openers) < 0) {
            this._openers.push(opener);
            setMouseTouchEvent($opener, ['touchend', 'click'], $.proxy(this.toggle, this), {
                namespace: this._id
            });
        }
    },

    /**
     * remove opener
     * @param {HTMLElement|jQuery|string} opener - element or selector
     */
    removeOpener: function(opener) {
        var $opener = $(opener);
        var index = util.inArray($opener[0], this._openers);

        if (index > -1) {
            $opener.off('.' + this._id);
            this._openers.splice(index, 1);
        }
    },

    /**
     * Open calendar with arranging position
     * @example
     * datepicker.open();
     */
    open: function() {
        var docEvents, evNamespace;
        if (this.isOpened() || !this._isEnabled) {
            return;
        }

        this._calendar.draw({
            date: this._date || new Date()
        });
        this._$element.show();

        if (!this.showAlways) {
            evNamespace = '.' + this._id;
            docEvents = 'touchstart' + evNamespace + ' mousedown' + evNamespace;
            $(document).on(docEvents, $.proxy(this._onMousedownDocument, this));
        }

        /**
         * @event DatePicker#open
         * @example
         * datePicker.on('open', function() {
         *     alert('open');
         * });
         */
        this.fire('open');
    },

    /**
     * Close calendar with unbinding some events
     * @exmaple
     * datepicker.close();
     */
    close: function() {
        var evNamespace;
        if (!this.isOpened()) {
            return;
        }
        evNamespace = '.' + this._id;
        $(document).off(evNamespace);
        this._$element.hide();

        /**
         * Close event - Datepicker
         * @event DatePicker#close
         * @example
         * datePicker.on('close', function() {
         *     alert('close');
         * });
         */
        this.fire('close');
    },

    /**
     * Toggle
     * @example
     * datepicker.toggle();
     */
    toggle: function() {
        var isOpened = this.isOpened();

        if (isOpened) {
            this.close();
        } else {
            this.open();
        }
    },

    /**
     * Get date-object of current Datepicker instance.
     * @returns {?Date} - Date
     * @example
     * // 2015-04-13
     * datepicker.getDate(); // new Date(2015, 3, 13)
     */
    getDate: function() {
        if (!this._date) {
            return null;
        }

        return new Date(this._date);
    },

    /**
     * Set date from values(year, month, date) and then fire 'update' custom event
     * @param {Date|number} date - Date instance or timestamp
     * @example
     * datepicker.setDate(new Date()); // Set today
     */
    setDate: function(date) {
        var prevDate, newDate, shouldUpdate;
        if (!(util.isDate(date) || util.isNumber(date)) || !dateUtil.isValidDate(new Date(date))) {
            return;
        }

        prevDate = this._date;
        newDate = new Date(date);
        shouldUpdate = (!prevDate || (newDate.getTime() !== prevDate.getTime())) && this.isSelectable(newDate);

        if (shouldUpdate) {
            this._date = newDate;
            this._syncToInput();
            this._calendar.draw({date: newDate});

            /**
             * Change event
             * @event DatePicker#change
             */
            this.fire('change');
        } else {
            this._syncToInput();
        }
    },

    /**
     * Set or update date-form
     * @param {String} [format] - date-format
     * @example
     * datepicker.setDateFormat('yyyy-MM-dd');
     * datepicker.setDateFormat('MM-dd, yyyy');
     * datepicker.setDateFormat('y/M/d');
     * datepicker.setDateFormat('yy/MM/dd');
     */
    setDateFormat: function(format) {
        if (!format) {
            return;
        }
        this._formatter = new DateFormatter(format);

        if (this._date) {
            this.setDate(this._date);
        }
    },

    /**
     * Return whether the calendar is opened or not
     * @returns {boolean} - true if opened, false otherwise
     * @example
     * datepicker.close();
     * datepicker.isOpened(); // false
     *
     * datepicker.open();
     * datepicker.isOpened(); // true
     */
    isOpened: function() {
        return this._$element.css('display') !== 'none';
    },

    /**
     * Returns timepicker instance
     * @returns {?Timepicker} - Timepicker instance
     * @example
     * var timepicker = this.getTimepicker();
     */
    getTimepicker: function() {
        return this._timepicker;
    },

    /**
     * Returns calendar instance
     * @returns {Calendar}
     */
    getCalendar: function() {
        return this._calendar;
    },

    /**
     * Set input element of this instance
     * @param {HTMLElement|jQuery} element - input element
     */
    setInput: function(element) {
        var $currentEl = this._$input;
        var $newEl = $(element);
        var namespaceSelector = '.' + this._id;

        $currentEl.off(namespaceSelector);
        $newEl.on('change' + namespaceSelector, $.proxy(this._onChangeInput, this));
        setMouseTouchEvent($newEl, ['touchend', 'click'], $.proxy(this.open, this), {
            namespace: this._id
        });
        this._$input = $newEl;
        this._syncToInput();
    },

    /**
     * Enable picker
     * @example
     * datepicker.disable();
     * datepicker.enable();
     */
    enable: function() {
        if (this._isEnabled) {
            return;
        }
        this._isEnabled = true;

        this._$input.removeAttr('disabled');
        util.forEach(this._openers, function(opener) {
            $(opener).removeAttr('disabled');
            setMouseTouchEvent(opener, ['touchend', 'click'], $.proxy(this.toggle, this), {
                namespace: this._id
            });
        }, this);
    },

    /**
     * Disable picker
     * @example
     * datepicker.enable();
     * datepicker.disable();
     */
    disable: function() {
        if (!this._isEnabled) {
            return;
        }

        this._isEnabled = false;
        this.close();

        this._$input.attr('disabled', true);
        util.forEach(this._openers, function(opener) {
            $(opener).attr('disabled', true).off('.' + this._id);
        }, this);
    },

    /**
     * Destroy - delete wrapper element and attach events
     */
    destroy: function() {
        var evNamespace = '.' + this._id;

        $(document).off(evNamespace);
        this._$input.off(evNamespace);
        this._$container.off(evNamespace);
        this._calendar.destroy();
        if (this._timepicker) {
            this._timepicker.destroy();
        }
        this._$element.remove();

        this._calendar
            = this._timepicker
            = this._$container
            = this._$element
            = this._$input
            = this._date
            = this._rangeModel
            = this._openers
            = this._isEnabled
            = this._formatter
            = this._id
            = null;
    }
});

util.CustomEvents.mixin(DatePicker);
module.exports = DatePicker;
