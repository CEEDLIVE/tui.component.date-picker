                var today = new Date();
            var datepicker = new tui.component.Datepicker('#wrapper-1', {
                type: 'date',
                date: null,
                selectableRanges: [
                    [new Date(2015, 8, 1), new Date(2016, 9, 3)],
                    [new Date(2016, 11, 22), new Date(2017, 0, 25)],
                    [today, new Date(today.getFullYear() + 1, today.getMonth(), 1)]
                ],
                openers: ['#datepicker-opener-1'],
                calendar: {
                    hasTodayBox: false
                },
                timepicker: {
                    showMeridiem: true
                },
                input: {
                    element: '#datepicker-input-1',
                    format: 'yyyy-MM-dd, hh:mm A'
                }
            });

            var datepicker2 = new tui.component.Datepicker('#wrapper-2', {
                input: {
                    element: '#datepicker-input-2',
                    format: 'yyyy-MM-dd, hh:mm A'
                },
                timepicker: {
                    showMeridiem: true,
                    style: 'selectbox'
                },
                calendar: {
                    hasJumpButtons: true
                },
                inlineTimepicker: true,
                type: 'date',
                date: new Date(2017, 1, 22),
                selectableRanges: [
                    [new Date(2015, 8, 1), new Date(2016, 9, 3)],
                    [new Date(2016, 11, 22), new Date(2017, 0, 25)],
                    [today, new Date(today.getFullYear() + 1, today.getMonth(), 1)]
                ],

            });
