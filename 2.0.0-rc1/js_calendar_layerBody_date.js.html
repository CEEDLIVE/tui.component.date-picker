'use strict';

var bodyTmpl = require('./../../../template/calendar/dateLayer.hbs');
var Base = require('./base');

var LAYER_TYPE_DATE = require('../constants').LAYER_TYPE_DATE;

var DATE_SELECTOR = '.tui-calendar-date';

/**
 * @class
 * @extends Base
 * @param {string} language - Initial language
 * @ignore
 */
var DateLayer = tui.util.defineClass(Base, {
    init: function(language) {
        Base.call(this, language);
    },

    /**
     * Layer type
     * @type {string}
     * @private
     */
    _type: LAYER_TYPE_DATE,

    /**
     * @returns {object} Template context
     * @override
     * @private
     */
    _makeContext: function(date) {
        var daysShort = this._localeText.titles.D;
        var year, month;

        date = date || new Date();
        year = date.getFullYear();
        month = date.getMonth() + 1;

        return {
            Sun: daysShort[0],
            Mon: daysShort[1],
            Tue: daysShort[2],
            Wed: daysShort[3],
            Thu: daysShort[4],
            Fri: daysShort[5],
            Sat: daysShort[6],
            year: year,
            month: month
        };
    },

    /**
     * Render date-layer
     * @param {Date} date - Date to render
     */
    render: function(date) {
        var context = this._makeContext(date);

        this._$element = $(bodyTmpl(context));
    },

    /**
     * Retunrs date elements
     * @returns {jQuery}
     */
    getDateElements: function() {
        return this._$element.find(DATE_SELECTOR);
    }
});

module.exports = DateLayer;
