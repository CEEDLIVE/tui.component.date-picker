    • DateRange-picker example
    Example
    var rangepicker = new DateRangePicker({
     startpicker: {
         input: '#start-input',
         container: '#start-container'
     },
     endpicker: {
         input: '#end-input',
         container: '#end-container'
     },
     type: 'date',
     format: 'yyyy-MM-dd'
     selectableRanges: [
         [new Date(2017, 3, 1), new Date(2017, 5, 1)],
         [new Date(2017, 6, 3), new Date(2017, 10, 5)]
     ]
 });
