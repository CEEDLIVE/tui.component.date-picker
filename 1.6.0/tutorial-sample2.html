    DatePicker - Customizing with options

    • Result
    • JS
    • HTML 
                var calendar = new tui.component.Calendar({
                element: '#layer'
            });

            var timePicker = new tui.component.TimePicker({
                showMeridian: true,
                defaultHour: 13,
                defaultMinute: 24
            });

            var datepicker1 = new tui.component.DatePicker({
                element: '#datepicker-default'
            }, calendar);

            var datepicker2 = new tui.component.DatePicker({
                element: '#datepicker-options',
                dateFormat: 'yyyy년 mm월 dd일 - ',
                selectableRanges: [
                    [{year: 2016, month: 9, date: 3}, {year: 2016, month: 9, date: 10}],
                    [{year: 2016, month: 9, date: 8}, {year: 2016, month: 9, date: 18}],
                    [{year: 2016, month: 10, date: 1}, {year: 2016, month: 10, date: 26}]
                ],
                openers: ['#opener'],
                timePicker: timePicker
            }, calendar);

            $('#disable').on('click', function() {
                datepicker2.disable();
            });

            $('#enable').on('click', function() {
                datepicker2.enable();
            });

            $('#add-range').on('click', function() {
                var startDate = {year: 2017, month: 8, date: 27};
                var endDate = {year: 2017, month: 9, date: 28};

                datepicker2.addRange(startDate, endDate);
            });

            $('#remove-range').on('click', function() {
                var startDate = {year: 2016, month: 9, date: 8};
                var endDate = {year: 2016, month: 9, date: 18};

                datepicker2.removeRange(startDate, endDate);
            });

            datepicker2.on('update', function() {
                console.log('?');
            });
